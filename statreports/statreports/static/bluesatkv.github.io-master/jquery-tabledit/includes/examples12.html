// Example #12
$('#example12').Tabledit({
  url: 'ajax/example.php',
  autoFocus: false,
  columns: {
    identifier: [0, 'id'],
    editable: [
      [1, 'make'],
      [2, 'year'],
      [3, 'model'],
      [4, 'platform']
    ]
  },
  onDraw: function() {
    // Select all inputs of second column and apply datepicker each of them
    $('table tr td:nth-child(4) input').each(function() {
      $(this).datepicker({
        format: 'yyyy-mm-dd',
        todayHighlight: true
      });
    });
  }
});
